const App = () => {
  const [count, setCount] = React.useState(0);
  
  React.useLayoutEffect(() => {
    console.log('1')
    
    return () => {
      console.log('2')
    }
  }, [count])
  
  React.useEffect(() => {
    console.log('3')
    
    return () => {
      console.log('4')
    }
  }, [count]);

  return (
      <div onClick={() => setCount(2)}>{count}</div>
  );
}

ReactDOM.render(<App/>, document.getElementById('app'));
